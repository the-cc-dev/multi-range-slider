Create the simplest multy-range slider component

HTML

<component class='multiRange'></component>

JAVASCRIPT

var multiRangeElm = document.querySelectorAll('.multiRange')[0];
var multiRange = new MultiRange(multiRangeElm, {
    ranges : [15, 44, 77, 88],
    step   : 0
});

// listen to custom 'remove' tag event
multiRange.on('change', onRangeChange);
multiRange.on('changed', onRangeChanged);

function onRangeChange(e){
    console.log(e, e.detail);
}

function onRangeChanged(e){
    console.log(e, e.detail);
}

A more complex example, with more ranges and inline html settings as custom data attributes

HTML

<component class='multiRange type1' data-min='150' data-max='5000' data-step='100'></component>

JAVASCRIPT

var multiRangeElm = document.querySelectorAll('.multiRange')[1];
var multiRange = new MultiRange(multiRangeElm, {
    ticks    : 80,
    step     : 10,
    minRange : 50
});
